        async function getReadingFromAPI(cards) {

            const cardData = cards.map(c => `Card: ${c.name} - Meaning: ${c.meaning}`).join('\n');

            const prompt = `Based on the following three tarot cards and their meanings, provide a short, mystical, and insightful three-sentence reading that connects their meanings together. Do not just list the meanings again. The first card represents the past, the second the present, and the third the future.


            Cards:

            ${cardData}

            `;

            

            const chatHistory = [];

            chatHistory.push({ role: "user", parts: [{ text: prompt }] });

            const payload = { contents: chatHistory };

            const apiKey = "" 

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;


            try {

                let response = null;

                let result = null;

                const maxRetries = 5;

                const initialDelay = 1000;

                

                for (let i = 0; i < maxRetries; i++) {

                    response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    

                    if (response.ok) {

                        result = await response.json();

                        break;

                    } else if (response.status === 429) {

                        // Exponential backoff

                        const delay = initialDelay * Math.pow(2, i);

                        await new Promise(res => setTimeout(res, delay));

                    } else {

                        throw new Error(`API call failed with status: ${response.status}`);

                    }

                }


                if (!response.ok) {

                    throw new Error("Maximum retries exceeded or non-recoverable error occurred.");

                }


                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    return result.candidates[0].content.parts[0].text;

                } else {

                    return "The threads of fate are intertwined in a way that words cannot describe. Seek deeper within yourself for the answer.";

                }

            } catch (error) {

                console.error("Error fetching reading:", error);

                return "The cosmic connection is weak, and the cards cannot be read at this moment. Please try again later.";

            }

        }


        async function drawCards() {

            // Disable button and show loading state

            drawButton.disabled = true;

            readingContainer.classList.add('hidden');

            cardContainer.innerHTML = '';

            readingContainer.innerHTML = '';

            loadingContainer.style.display = 'flex';


            // Randomly select 3 unique cards

            const shuffledDeck = deck.sort(() => 0.5 - Math.random());

            const selectedCards = shuffledDeck.slice(0, 3);

            const isReversed = Math.random() < 0.3; // 30% chance for a reversed card


            // Create and display the cards with a flip animation

            selectedCards.forEach((card, index) => {

                const cardDiv = document.createElement('div');

                cardDiv.className = `tarot-card ${isReversed && index === 1 ? 'reversed' : ''}`;

                

                const cardInner = document.createElement('div');

                cardInner.className = 'card-inner';

                

                const cardFront = document.createElement('div');

                cardFront.className = 'card-front';

                cardFront.innerHTML = `

                    <div class="card-position">${positions[index]}</div>

                    <div class="card-image">${getCardImageSVG(card.name)}</div>

                    <div class="card-name">${card.name}</div>

                `;


                const cardBack = document.createElement('div');

                cardBack.className = 'card-back';

                cardBack.innerHTML = cardBackSVG;


                cardInner.appendChild(cardBack);

                cardInner.appendChild(cardFront);

                cardDiv.appendChild(cardInner);

                cardContainer.appendChild(cardDiv);


                // Add flip animation after a small delay

                setTimeout(() => {

                    cardDiv.classList.add('flipped');

                }, 500 * index);

            });


            // Get reading from API

            const readingText = await getReadingFromAPI(selectedCards);


            // Display reading

            loadingContainer.style.display = 'none';

            readingContainer.classList.remove('hidden');

            readingContainer.innerHTML = `

                <h2>The Reading</h2>

                <div class="reading-card-meaning">

                    <h3>${positions[0]}: ${selectedCards[0].name}</h3>

                    <p>${selectedCards[0].meaning}</p>

                </div>

                <div class="reading-card-meaning">

                    <h3>${positions[1]}: ${selectedCards[1].name}</h3>

                    <p>${selectedCards[1].meaning}</p>

                </div>

                <div class="reading-card-meaning">

                    <h3>${positions[2]}: ${selectedCards[2].name}</h3>

                    <p>${selectedCards[2].meaning}</p>

                </div>

                <h3>In Conjunction:</h3>

                <p>${readingText}</p>

            `;


            drawButton.disabled = false;

        }


        drawButton.addEventListener('click', drawCards);

    </script>
